import gwfvis

# load default vis configurations
vis_config = gwfvis.load_default_vis_config()

# add a default map layer
gwfvis.add_plugin(vis_config, 'hidden', gwfvis.DEFAULT_SATELITTE_LAYER)

# set a view for the map
vis_config['view'] = {
    'center': [51.312588, -116.021118],
    'zoom': 10
}

# show the visualization
gwfvis.generate_vis(vis_config)
